File: lib/widget/yo-group.scss

@charset "utf-8";
@import "classes";
@import "../fragment/yo-list";
@import "yo-index";

/**
 * @class widget
 */

/**
 * 构造列表的自定义使用方法
 -----------------------
 * 使用方法，详见 [Demo](http://doyoe.github.io/Yo/demo/widget/yo-group.html)
 * @method yo-group
 * @param {default | String} $name 定义yo-group名称
 * @param {default | Length} $top 定义yo-group顶部偏移值
 */

@mixin yo-group(
    $name: default,
    $top: default) {
    // 区别是否新增实例还是修改本身
    @if $name == default {
        $name: "";
    } @else {
        $name: "-#{$name}";
    }
    // 如果值为default，则取config的定义
    @if $top == default {
        $top: map-get($group, top);
    }
    .yo-group#{$name} {
        @if $top != map-get($group, top) {
            top: $top;
        }
    }
}

// 调用本文件时载入group基础构造
.yo-group {
    @include _group;
}
