@charset "utf-8";
@import "classes";

/**
 * document: doc
 * Author: joy.du(飘零雾雨)
 * Time: 2012.1.1
 ------------------------
 * @method doc
 * @param {String} $name 定义page-name
 * @param {String} $fixed 定义头尾是否固定
 * @param {Length} $hd-gap 定义主体到viewport顶部的距离
 * @param {Length} $ft-gap 定义主体到viewport底部的距离
 */

@include _doc;
@mixin doc(
    $name: default,
    $is-scroll: default) {
    // 区别是否新增实例还是修改本身
    @if $is-scroll == default {
        $is-scroll: map-get($doc, is-scroll);
    }

    // 如果$is-scroll不等于config定义
    @if $is-scroll != map-get($doc, is-scroll) {
        @if $is-scroll == false {
            html,body {
                overflow: hidden;
                height: 100%;
            }
        } @else {
            html,body {
                overflow: visible;
                height: auto;
            }
        }
    }
}
